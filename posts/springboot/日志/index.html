spring boot 日志配置

POM文件引入依赖

<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter</artifactId>
</dependency>

1.修改日志输出级别

debug: true
trace: true
  • debug 模式会将INFO 、ERROR、WARN 信息给输出
  • trace 级别会更详细，能囊括springboot的全部log输出

还可以设置其它的日志系统

logging:
  level:
    root: warn

2.使用彩色日志编码输出

spring:
  output:
    ansi:
      enabled: always

3. 传统方式配置logback

新建文件conf/logback-prod.xml

<configuration>
    <!-- 控制台打印日志的相关配置 -->
    <appender name="STDOUT" class="ch.qos.logback.core.ConsoleAppender">
        <!-- 日志格式 -->
        <encoder>
            <pattern>%d{yyyy-MM-dd HH:mm:ss} %highlight(%-4relative) [%thread] %highlight(%-5level) %logger{35} - %msg %n</pattern>
        </encoder>
    </appender>

    <appender name="FILE" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <!-- 保存日志文件的路径 -->
        <file>.logs/log.log</file>
        <!-- 日志格式 -->
        <encoder>
            <pattern>%d{yyyy-MM-dd HH:mm:ss} %highlight(%-4relative) [%thread] %-5level %logger{35} - %msg %n</pattern>
        </encoder>
        <!-- 循环政策：基于时间创建日志文件 -->
        <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
            <!-- 日志文件名格式 -->
            <fileNamePattern>.logs/%d{yyyy-MM-dd}.log</fileNamePattern>
            <!-- 最大保存时长-->
            <maxHistory>10</maxHistory>
            <!--日志文件最大大小-->
            <maxFileSize>10MB</maxFileSize>
            <!--日志上限-->
            <totalSizeCap>1GB</totalSizeCap>
            <!---在启动的时候清除历史日志-->
            <cleanHistoryOnStart>false</cleanHistoryOnStart>
        </rollingPolicy>
    </appender>

    <!--处理日志-->
    <root level="INFO">
        <appender-ref ref="STDOUT" />
        <appender-ref ref="FILE"/>
    </root>
</configuration>

然后在application-prod.yaml中写入

logging:
  config: classpath:conf/logback-prod.xml

最后可以在 application.yaml 指定启动的profile

spring:
  profiles:
    active: prod

###4. 使用springProfile配置logback

首先新建/conf/logback-spring.xml，然后在application.yaml 指定

logging:
  config: classpath:conf/logback-spring.xml

<configuration>
    <!--可以引入环境变量-->
    <springProperty scope="context" name="FILE_PATH" source="logging.path"
                    defaultValue=".logs"/>
    <!--开发环境配置-->
    <springProfile name="dev">
        <appender name="STDOUT" class="ch.qos.logback.core.ConsoleAppender">
            <!-- 日志格式 -->
            <encoder>
                <pattern>%d{yyyy-MM-dd HH:mm:ss} %highlight(%-4relative) [%thread] %highlight(%-5level) %logger{35} - %msg %n</pattern>
            </encoder>
        </appender>
        <root level="INFO">
            <appender-ref ref="STDOUT" />
        </root>
    </springProfile>

    <!--生产环境配置-->
    <springProfile name="prod">
        <appender name="FILE" class="ch.qos.logback.core.rolling.RollingFileAppender">
            <!-- 保存日志文件的路径 -->
            <file>${FILE_PATH}/log.log</file>
            <!-- 日志格式 -->
            <encoder>
                <pattern>%d{yyyy-MM-dd HH:mm:ss} %highlight(%-4relative) [%thread] %-5level %logger{35} - %msg %n</pattern>
            </encoder>
            <!-- 循环政策：基于大小和时间创建日志文件 -->
            <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
                <!-- 日志文件名格式 -->
                <fileNamePattern>${FILE_PATH}/%d{yyyy-MM-dd}_%i.log</fileNamePattern>
                <!-- 最大保存时长-->
                <maxHistory>10</maxHistory>
                <!--日志文件最大大小-->
                <maxFileSize>10MB</maxFileSize>
                <!--日志上限-->
                <totalSizeCap>1GB</totalSizeCap>
                <!---在启动的时候清除历史日志-->
                <cleanHistoryOnStart>false</cleanHistoryOnStart>
            </rollingPolicy>
        </appender>
        <root level="INFO">
            <appender-ref ref="FILE" />
        </root>
    </springProfile>
</configuration>

在上面我们可以用 springProperty 引入环境变量，之后我们就可以在 application.yaml中设置我们的环境变量

logging:
  config: classpath:conf/logback-spring.xml
  path: ./logs

5. 添加 logstash + elk 进行日志可视化

  1. 首先安装ELK:
'elsticsearch'
brew install elastic/tap/elasticsearch-full
'logstash'
brew install elastic/tap/logstash-full
'kibana'
brew install elastic/tap/kibana-full

  1. 然后配置logback-spring.xml

     <springProfile name="test">
        <appender name="STASH" class="net.logstash.logback.appender.LogstashTcpSocketAppender">
            <!--配置logStash 服务地址-->
            <destination>localhost:4560</destination>
            <!-- 日志输出编码 -->
            <encoder charset="UTF-8"
                     class="net.logstash.logback.encoder.LoggingEventCompositeJsonEncoder">
                <providers>
                    <timestamp>
                        <timeZone>UTC</timeZone>
                    </timestamp>
                    <pattern>
                        <pattern>
                            {
                            "logLevel": "%level",
                            "serviceName": "${springAppName:-}",
                            "pid": "${PID:-}",
                            "thread": "%thread",
                            "class": "%logger{40}",
                            "rest": "%message"
                            }
                        </pattern>
                    </pattern>
                </providers>
            </encoder>
        </appender>
        <root level="INFO">
            <appender-ref ref="STASH" />
        </root>
    </springProfile>

    然后添加logstash依赖

    <dependency>
    <groupId>net.logstash.logback</groupId>
    <artifactId>logstash-logback-encoder</artifactId>
    <version>5.3</version>
</dependency>

    3.接下来新建一个logstash.conf的配置文件，这个文件可以放在任意位置

    input {
  tcp {
    mode => "server"
    host => "localhost"
    port => 4560
    codec => json_lines
  }
}
output {
  elasticsearch {
    hosts => "localhost:9200"
    index => "springboot-logstash-%{+YYYY.MM.dd}"
  }
}

其中input代表logstash的配置信息，包括端口号什么的

然后output代表elasticsearch 的地址

  1. 最后依次启动服务

    elasticsearch
kibana
logstash -f logstash.conf

  2. 然后进入kibana的主页即可看到数据信息（前提我得会elaticsearch）

    http://localhost:5601/app/kibana#/discover
